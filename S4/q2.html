import { Component, OnInit } from '@angular/core'; import { HttpClient } from '@angular/common/http'; @Component({ selector: 'app-student-registration', templateUrl: './student-registration.component.html', styleUrls: ['./student-registration.component.css'] }) export class StudentRegistrationComponent implements OnInit { student = { fname: '', lname: '', age: null }; greeting: string = ''; responseMessage: string = ''; constructor(private http: HttpClient) {} ngOnInit(): void { const hour = new Date().getHours(); if (hour < 12) this.greeting = "Good Morning!"; else if (hour < 17) this.greeting = "Good Afternoon!"; else this.greeting = "Good Evening!"; } registerStudent() { this.http.post('https://jsonplaceholder.typicode.com/posts', this.student) .subscribe({ next: () => this.responseMessage = "Student Registered Successfully!", error: () => this.responseMessage = "Something went wrong!" }); } }
